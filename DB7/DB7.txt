FROM Mitarbeiter, Wis_Mitarbeiter WHERE	Mitarbeiter.PersonalID	=	Wis_Mitarbeiter.PersonalID;

CREATE OR REPLACE TRIGGER insert_WiMiInfo
  INSTEAD OF INSERT ON WiMiInfo
DECLARE
x INTEGER;
BEGIN
  SELECT COUNT(*) + 1
  INTO x
FROM Mitarbeiter;

END;
